    "class BankAccount:\n",
    "    def __init__(self, account_holder: str, balance: float = 0.0) -> None:\n",
    "        self.account_holder = account_holder\n",
    "        self.balance = balance\n",
    "\n",
    "    def deposit(self, amount: float) -> None:\n",
    "        if amount <= 0:\n",
    "            raise ValueError(\"Deposit amount must be greater than zero.\")\n",
    "        self.balance += amount\n",
    "\n",
    "    def withdraw(self, amount: float) -> None:\n",
    "        if amount <= 0:\n",
    "            raise ValueError(\"Withdrawal amount must be greater than zero.\")\n",
    "        if amount > self.balance:\n",
    "            raise ValueError(\"Insufficient balance.\")\n",
    "        self.balance -= amount\n",
    "\n",
    "    def get_balance(self) -> float:\n",
    "        return self.balance\n",
    "\n",
    "\n",
    "account = BankAccount(\"Alice\", 1000.0)\n",
    "account.deposit(250.0)\n",
    "account.withdraw(100.0)\n",
    "\n",
    "print(account.account_holder)\n",
    "print(account.get_balance())"

    "### Guidelines\n",

    "- Log useful context\n",
    "- Avoid logging secrets or sensitive personal data\n",

    "import logging\n",
    "\n",
    "logger = logging.getLogger(\"salary_calculator\")\n",
    "\n",
    "class EmployeeSalaryCalculator:\n",
    "    \"\"\"Calculate employee salary details.\"\"\"\n",
    "\n",
    "    TAX_RATE = 0.10\n",
    "\n",
    "    def __init__(self, employee_name: str, monthly_salary: float) -> None:\n",
    "        self.employee_name = employee_name\n",
    "        self.monthly_salary = monthly_salary\n",
    "\n",
    "    def calculate_net_salary(self) -> float:\n",
    "        \"\"\"\n",
    "        Calculate salary after tax deduction.\n",
    "\n",
    "        Returns:\n",
    "            Net monthly salary after tax.\n",
    "        \"\"\"\n",
    "        if self.monthly_salary < 0:\n",
    "            raise ValueError(\"Monthly salary cannot be negative.\")\n",
    "\n",
    "        tax_amount = self.monthly_salary * self.TAX_RATE\n",
    "        net_salary = self.monthly_salary - tax_amount\n",
    "\n",
    "        logger.info(\"Calculated net salary for employee=%s\", self.employee_name)\n",
    "        return net_salary\n",
    "\n",
    "\n",
    "calculator = EmployeeSalaryCalculator(\"Ravi\", 5000.0)\n",
    "print(\"Net salary:\", calculator.calculate_net_salary())"
